An Angular service containing a HTTP
Simply include a file named .deployment at the root of your repo, which contains the path to the folder you would like deployed to your Azure website.

// Service containing http call to API
var app = angular.module('AwardProvider', []);
app.factory('MediaService', function($rootScope, $http, $q) {
	return {
		getRecentMedia: function(location_id) {
			var deferred = $q.defer();

			// http post to api (may take a second or two)
			$http.post('/api/recent_media', { location_id: location_id}).success(function (data) {
				deferred.resolve(data); // resolve the promise
			});
			return deferred.promise; // returns promise to caller (with no data yet)
		}
	};
});
